# Poop Catcher Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single `index.html` browser game where 💩 falls from the sky and the player catches it with 🧻.

**Architecture:** One HTML file — Canvas 2D game loop via `requestAnimationFrame`, vanilla JS only. Three plain objects: `Game` (state/loop), `Poop` (falling entity), `Player` (catcher). No build tools, no dependencies.

**Tech Stack:** HTML5 Canvas, Vanilla JS (ES6), CSS (inline, minimal)

## File Structure

| File | Role |
|------|------|
| `index.html` | Entire game — markup, styles, all JS |

Everything lives in one `<script>` block. Build incrementally — each task leaves the file runnable in a browser.

### Task 2: State machine + start screen

Replace the `<script>` block content with:

```javascript
const canvas = document.getElementById('game');
const ctx = canvas.getContext('2d');

function resize() {
  canvas.width = window.innerWidth;
  canvas.height = window.innerHeight;
}
window.addEventListener('resize', resize);
resize();

// --- State ---
let state = 'start'; // 'start' | 'playing' | 'gameover'

// --- Draw ---
function draw() {
  ctx.fillStyle = '#1a1a2e';
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  if (state === 'start') drawStart();
  else if (state === 'playing') drawPlaying();
  else if (state === 'gameover') drawGameover();
}

function drawStart() {
  ctx.textAlign = 'center';
  ctx.font = 'bold 64px serif';
  ctx.fillStyle = '#fff';
  ctx.fillText('💩 POOP CATCHER 🧻', canvas.width / 2, canvas.height / 2 - 60);
  ctx.font = '28px serif';
  ctx.fillStyle = '#ccc';
  ctx.fillText('catch the poop before it hits the ground!', canvas.width / 2, canvas.height / 2);
  ctx.font = '22px serif';
  ctx.fillStyle = '#aaa';
  ctx.fillText('Press any key or tap to start', canvas.width / 2, canvas.height / 2 + 60);
}

function drawPlaying() {
  ctx.textAlign = 'center';
  ctx.font = '32px serif';
  ctx.fillStyle = '#fff';
  ctx.fillText('playing...', canvas.width / 2, canvas.height / 2);
}

function drawGameover() {
  ctx.textAlign = 'center';
  ctx.font = 'bold 72px sans-serif';
  ctx.fillStyle = '#ff4444';
  ctx.fillText('GAME OVER', canvas.width / 2, canvas.height / 2 - 40);
  ctx.font = '24px sans-serif';
  ctx.fillStyle = '#aaa';
  ctx.fillText('Press any key or tap to play again', canvas.width / 2, canvas.height / 2 + 40);
}

// --- Input ---
document.addEventListener('keydown', () => {
  if (state === 'start') state = 'playing';
  else if (state === 'gameover') state = 'start';
});
canvas.addEventListener('click', () => {
  if (state === 'start') state = 'playing';
  else if (state === 'gameover') state = 'start';
});

// --- Loop ---
function loop() {
  draw();
  requestAnimationFrame(loop);
}
requestAnimationFrame(loop);
```

```javascript
function collidesWithPlayer(poop) {
```

- [ ] **Step 3: Update updatePoops to handle catches**

Replace `updatePoops`:

```javascript
function updatePoops(timestamp) {
  spawnPoop(timestamp);
  poops = poops.filter(p => {
    p.y += p.speed;
    if (collidesWithPlayer(p)) {
      score++;
      scoreBounce = 1;
      effects.push({ x: p.x, y: canvas.height - 80, timer: 300 });
      updateDifficulty();
      return false;
    }
    return p.y < canvas.height + 50;
  });
}
```

- [ ] **Step 4: Update effects each frame**

```javascript
function updateEffects(dt) {

    e.timer -= dt;
    return e.timer > 0;
  });
}
```

```html
        if (collidesWithPlayer(p)) {
          score++;
          scoreBounce = 1;
          effects.push({ x: p.x, y: canvas.height - 80, timer: 300 });
          updateDifficulty();
          return false;
        }
        if (p.y >= canvas.height + 50) {
          lives--;
          flashTimer = 400;
          return false;
        }
        return true;
      });
    }

    function updateEffects(dt) {
      if (scoreBounce > 0) scoreBounce = Math.max(0, scoreBounce - 0.06);
      if (flashTimer > 0) flashTimer = Math.max(0, flashTimer - dt);
      effects = effects.filter(e => { e.timer -= dt; return e.timer > 0; });
      if (lives <= 0) state = 'gameover';
    }

    // --- Start / Reset ---
    function startGame() {
      score = 0;
      lives = 3;
      poops = [];
      effects = [];
      flashTimer = 0;
      scoreBounce = 0;
      speedMultiplier = 1;
      spawnInterval = 2000;
      lastSpawnTime = 0;
      player.x = canvas.width / 2;
      initClouds();
      state = 'playing';
    }

    // --- Draw ---
    function drawStart() {
      ctx.textAlign = 'center';
      ctx.font = 'bold 64px serif';
      ctx.fillStyle = '#fff';
      ctx.fillText('💩 POOP CATCHER 🧻', canvas.width / 2, canvas.height / 2 - 60);
      ctx.font = '28px serif';
      ctx.fillStyle = '#ccc';
      ctx.fillText('catch the poop before it hits the ground!', canvas.width / 2, canvas.height / 2);
      ctx.font = '22px serif';
      ctx.fillStyle = '#aaa';
      ctx.fillText('Press any key or tap to start', canvas.width / 2, canvas.height / 2 + 60);
    }

    function drawPlaying() {
      ctx.textAlign = 'center';
      ctx.font = '32px serif';
      poops.forEach(p => ctx.fillText('💩', p.x, p.y));
      ctx.font = '48px serif';
      ctx.fillText('🧻', player.x, canvas.height - 40);
      ctx.font = '28px serif';
      effects.forEach(e => ctx.fillText('💥', e.x, e.y));
      const scale = 1 + scoreBounce * 0.5;
      ctx.save();
      ctx.textAlign = 'left';
      ctx.font = `bold ${Math.round(32 * scale)}px sans-serif`;
      ctx.fillStyle = '#fff';
      ctx.fillText(`Score: ${score}`, 20, 50);
      ctx.restore();
      ctx.textAlign = 'right';
      ctx.font = '28px serif';
      ctx.fillText('❤️'.repeat(Math.max(0, lives)), canvas.width - 20, 50);
    }

    function drawGameover() {
      ctx.textAlign = 'center';
      ctx.font = 'bold 72px sans-serif';
      ctx.fillStyle = '#ff4444';
      ctx.fillText('GAME OVER', canvas.width / 2, canvas.height / 2 - 60);
      ctx.font = '36px sans-serif';
      ctx.fillStyle = '#fff';
      ctx.fillText(`Score: ${score}`, canvas.width / 2, canvas.height / 2);
      ctx.font = '24px sans-serif';
      ctx.fillStyle = '#aaa';
      ctx.fillText('Press any key or tap to play again', canvas.width / 2, canvas.height / 2 + 60);
    }

    function draw() {
      ctx.fillStyle = '#1a1a2e';
      ctx.fillRect(0, 0, canvas.width, canvas.height);
      if (flashTimer > 0) {
        ctx.fillStyle = `rgba(255, 0, 0, ${(flashTimer / 400) * 0.4})`;
        ctx.fillRect(0, 0, canvas.width, canvas.height);
      }
      ctx.font = '40px serif';
      clouds.forEach(c => ctx.fillText('☁️', c.x, c.y));
      if (state === 'start') drawStart();
      else if (state === 'playing') drawPlaying();
      else if (state === 'gameover') drawGameover();
    }

    // --- Input ---
    document.addEventListener('keydown', e => {
      keys[e.key] = true;
      if (state === 'start') startGame();
      else if (state === 'gameover') startGame();
    });
    document.addEventListener('keyup', e => { keys[e.key] = false; });
    canvas.addEventListener('mousemove', e => {
      if (state === 'playing') player.x = e.clientX;
    });
    canvas.addEventListener('touchmove', e => {
      e.preventDefault();
      if (state === 'playing') player.x = e.touches[0].clientX;
    }, { passive: false });
    canvas.addEventListener('click', () => {
      if (state === 'start') startGame();
      else if (state === 'gameover') startGame();
    });
    canvas.addEventListener('touchstart', () => {
      if (state === 'start') startGame();
      else if (state === 'gameover') startGame();
    });

    // --- Loop ---
    let lastTimestamp = 0;
    function loop(timestamp) {
      const dt = timestamp - lastTimestamp;
      lastTimestamp = timestamp;
      updateClouds();
      if (state === 'playing') {
        updatePlayer();
        updatePoops(timestamp);
        updateEffects(dt);
      }
      draw();
      requestAnimationFrame(loop);
    }
    requestAnimationFrame(loop);
  </script>
</body>
</html>
```
